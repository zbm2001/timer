Quick start

// 创建一个定时器对象
var timer = new Timer();

// 执行一个延时任务：
timer.setTimeout('do', function(){ console.log('do setTimeout') }, 2 * 1000)

// 执行一个定时任务：
timer.setInterval('do', function(){ console.log('do setInterval') }, 2 * 1000)

// 清除延时任务:
timer.clearTimeout('do');

// 清除定时任务:
timer.clearInterval('do');
